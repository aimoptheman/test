📄 Component Source

import React, { useState, useCallback } from 'react';
import './MultiStateButton.css';

export function MultiStateButton({
  states,
  initialIndex = 0,
  onStateChange,
  className,
  style,
}: MultiStateButtonProps): React.JSX.Element {
  const [index, setIndex] = useState(initialIndex);
  const current = states[index];
  const isImage =
    current.content.startsWith('http://') ||
    current.content.startsWith('https://');

  const advance = useCallback(() => {
    setIndex(prev => {
      const next = (prev + 1) % states.length;
      onStateChange?.({ index: next, state: states[next] });
      return next;
    });
  }, [states, onStateChange]);

  return (
    <button
      type="button"
      className={['msb', className].filter(Boolean).join(' ')}
      aria-label={current.label ?? current.content}
      onClick={advance}
      style={style}
    >
      {isImage ? (
        <img src={current.content} alt={current.label ?? ''} className="msb-img" />
      ) : (
        <span>{current.content}</span>
      )}
    </button>
  );
}

export interface MsbState {
  /** Emoji, plain text, or an https:// image URL. */
  content: string;
  /** Used for aria-label; falls back to content. */
  label?: string;
}

export interface MsbStateChangeEvent {
  index: number;
  state: MsbState;
}

export interface MultiStateButtonProps {
  states: MsbState[];
  initialIndex?: number;
  onStateChange?: (event: MsbStateChangeEvent) => void;
  className?: string;
  style?: React.CSSProperties;
}

import { MultiStateButton } from '@aimoptheman/react-components';

function App() {
  const moods = [
    { content: '😊', label: 'Happy'   },
    { content: '😐', label: 'Neutral' },
    { content: '😢', label: 'Sad'     },
    { content: '😠', label: 'Angry'   },
    { content: '🤩', label: 'Excited' },
  ];

  return (
    <MultiStateButton
      states={moods}
      onStateChange={({ index, state }) =>
        console.log(`State ${index}: ${state.label}`)
      }
    />
  );
}
