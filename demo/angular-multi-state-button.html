📄 Component Source

import {
  ChangeDetectionStrategy, Component, EventEmitter,
  Input, OnInit, Output, computed, signal,
} from '@angular/core';
import { NgIf } from '@angular/common';
import { MsbState, MsbStateChangeEvent } from './multi-state-button.types';

@Component({
  selector: 'msb-multi-state-button',
  standalone: true,
  imports: [NgIf],
  templateUrl: './multi-state-button.component.html',
  styleUrls: ['./multi-state-button.component.css'],
  changeDetection: ChangeDetectionStrategy.OnPush,
})
export class MultiStateButtonComponent implements OnInit {
  /** Ordered list of states. Required; must be non-empty. */
  @Input({ required: true })
  set states(v: MsbState[]) { this._states.set(v); }
  get states(): MsbState[] { return this._states(); }

  /** Zero-based index of the initial state. */
  @Input()
  set initialIndex(v: number) { this._index.set(v); }
  get initialIndex(): number { return this._index(); }

  /** Emits after each click with the new index and state. */
  @Output() readonly stateChange = new EventEmitter<MsbStateChangeEvent>();

  private readonly _states = signal<MsbState[]>([]);
  private readonly _index  = signal(0);

  readonly currentState = computed(() => this._states()[this._index()]);
  readonly isImageContent = computed(() => {
    const c = this.currentState()?.content ?? '';
    return c.startsWith('http://') || c.startsWith('https://');
  });

  ngOnInit(): void {
    if (!this._states().length)
      throw new Error('msb-multi-state-button: [states] must be non-empty.');
  }

  advance(): void {
    const next = (this._index() + 1) % this._states().length;
    this._index.set(next);
    this.stateChange.emit({ index: next, state: this.currentState() });
  }
}
<button
  class="msb"
  type="button"
  (click)="advance()"
  [attr.aria-label]="currentState()?.label ?? currentState()?.content"
>
  <img
    *ngIf="isImageContent()"
    [src]="currentState()!.content"
    [alt]="currentState()?.label ?? ''"
    class="msb-img"
  />
  <span *ngIf="!isImageContent()">{{ currentState()?.content }}</span>
</button>
:host { display: inline-block; }

🚀 Installation & Usage

$ npm install @aimoptheman/angular-components 
import { Component } from '@angular/core';
import { MultiStateButtonComponent } from '@aimoptheman/angular-components';

@Component({
  standalone: true,
  imports: [MultiStateButtonComponent],
  template: `
    <msb-multi-state-button
      [states]="moods"
      (stateChange)="onMoodChange($event)"
    />
  `
})
export class AppComponent {
  moods = [
    { content: '😊', label: 'Happy'    },
    { content: '😐', label: 'Neutral'  },
    { content: '😢', label: 'Sad'      },
    { content: '😠', label: 'Angry'    },
    { content: '🤩', label: 'Excited'  },
  ];

  onMoodChange({ index, state }) {
    console.log(`State changed to index ${index}: ${state.label}`);
  }
}
